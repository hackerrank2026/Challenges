class AutocorrectTextarea extends React.Component {
  constructor(props) {
    super(props);
    this.state = { text: "" };
  }

  handleChange = (e) => {
    const input = e.target.value;
    const { corrections } = this.props;

    if (input.endsWith(" ")) {
      const newVal = input.slice(0, -1);
      const words = newVal.split(" ");
      const lastWord = words[words.length - 1];
      if (corrections && corrections[lastWord]) {
        words[words.length - 1] = corrections[lastWord];
      }

      this.setState({ text: words.join(" ") + " " });
    } else {
      this.setState({ text: input });
    }
  };

  render() {
    return (
      <div className="text-center">
        <textarea
          value={this.state.text}
          onChange={this.handleChange}
          data-testid="textarea"
          rows={10}
          cols={80}
          className="card"
        />
      </div>
    );
  }
}

  const[category, setCategory]=useState("-");
  const[weight, setWeight]=useState("")
  const[height, setHeight]=useState("")
  const[unit, setUnit] = useState("metric")


  const calculateBMI=()=>{
    if(!height.trim() ||!weight.trim() ||Number(height)<=0 || Number(weight)<=0){
      return;
    }

    if(unit==="metric"){
      let hh=Number(height)/100;
      const calc=Number(weight)/(hh*hh);
      setBmi(calc.toFixed(2))
      if(calc<18.5)
        setCategory("Underweight")
      else if(calc>=18.5 && calc<=24.9)
        setCategory("Normal weight")
      else if (calc >=25 && calc <=29.9)
        setCategory("Overweight")
      else if(calc>=30)
        setCategory("Obesity")
    }
    else if(unit==="imperial"){
      const hh =Number(height) *Number(height)
      const calc=703*Number(weight)/hh;
      setBmi(calc.toFixed(2))
      if(calc<18.5)
        setCategory("Underweight")
      else if(calc>=18.5 && calc<=24.9)
        setCategory("Normal weight")
      else if (calc >=25 && calc <=29.9)
        setCategory("Overweight")
      else if(calc>=30)
        setCategory("Obesity")
    }


  }

  return (
    <div className="layout-column align-items-center justify-content-start bmi-calculator" data-testid="bmi-calculator">

import React, { useEffect, useState } from 'react';
import stocks from "../../stocks";

function Users() {
  const [data, setData] = useState(stocks);
  const [currMonth, setCurrMonth] = useState("January");
  const [monthData, setMonthData] = useState([]);
  const [totalProfit, setTotalProfit] = useState(0);
  const [totalLoss, setTotalLoss] = useState(0);

  const months = [
    "January", "February", "March", "April", "May", "June",
    "July", "August", "September", "October", "November", "December"
  ];

  useEffect(() => {
    const filteredMonth = data.filter(f => f.date.includes(currMonth))
    setMonthData(filteredMonth);
    total_profit_Loss(filteredMonth); // to avoiding the stale condition we have to pass filteredMonth here.
  }, [currMonth])

  const handleSort = () => {
    const sortedData = [...monthData].sort((a, b) => (b.close - b.open)-(a.close - a.open));
    setMonthData(sortedData);
  }

  function total_profit_Loss(filteredMonth) {  //for handling stale condition we are accepting "filteredMonth"  instead of direct "monthData"
    let temp = 0;
    let total_profit = 0;
    let total_loss = 0;
    for (let i = 0; i < filteredMonth.length; i++) {
      temp = (filteredMonth[i].close - filteredMonth[i].open);
      if (temp < 0) {
        total_loss += temp;
      }
      else {
        total_profit += temp;
      }
    }
    setTotalProfit(total_profit);
    setTotalLoss(total_loss);
  }

  return (
    <>
      <div className='layout-row justify-content-center'>
        <select className='mt-13 mr-5' data-testid="select-month" defaultValue={'January'} onChange={(e) => { setCurrMonth(e.target.value) }}>
          {months.map(m =>
            <option>{m}</option>
          )}
        </select>
        <button data-testid="sort" onClick={handleSort}>Sort</button>
      </div>
      <div className='layout-row wrap w-100 justify-content-center' data-testid="stocks">

        {data && monthData.map((stock, index) =>
          <div className='card layout-column w-20 ma-10 px-8' key={index}>
            <p className='ma-0'>Date: {stock.date}</p>
            <p className='ma-0'>High: {stock.high}</p>
            <p className='ma-0'>Low: {stock.low}</p>
            <p className='ma-0'>Profit: {((stock.close - stock.open) < 0) ? "❌" : "✅"} ({Number(Math.abs((stock.close - stock.open))).toFixed(2)})</p>
          </div>
        )
        }

      </div>
      <div className='layout-column justify-content-center align-items-center' >
        <p className='mb-0 font-weight-bold' data-testid='total-profit'>Total Profit: <span style={{ color: "#1ba94c" }}>{totalProfit.toFixed(2)}</span></p>
        <p className='font-weight-bold' data-testid='total-loss'>Total Loss: <span style={{ color: "red" }}>{totalLoss.toFixed(2)}</span></p>
      </div>
    </>
  )
}

export default class Navigation extends Component {
  constructor(props) {
    super(props);
    this.state = {
      locations: props.locations,
    };
  }

  // Used for rendering
  getClasses(ctx, index) {
    let classes = `material-icons ${ctx}`;
    if (ctx === "dots") {
      if (this.isLast(index)) {
        classes += " hidden";
      }
    } else {
      classes += this.isLast(index) ? " small" : " x-small";
      if (index === 0) {
        classes += " first";
      }
    }
    return classes;
  }

  // Used for rendering
  isLast(index) {
    return index === this.props.locations.length - 1;
  }

  moveUp = (index) => {
    const data = [...this.state.locations];
    const temp = data[index];
    data[index] = data[index - 1];
    data[index - 1] = temp;
    this.setState({ locations: data });
  };

  moveDown = (index) => {
    const data = [...this.state.locations];
    const temp = data[index];
    data[index] = data[index + 1];
    data[index + 1] = temp;
    this.setState({ locations: data });
  };

  render() {
    const { locations } = this.state;
    return (
      <div className="layout-row align-items-center justify-content-center navigation-screen">
        <div className="card layout-row flat map-card">
          <section className="card pb-16 pr-16 flex-auto layout-column justify-content-center">

const ImagePreview = ({ images }) => {
  const [items, setItems] = useState(images);

  const imgToggle = (index) => {
    setItems((prev) =>
      prev.map((img, i) =>
        i === index ? { ...img, visible: !img.visible } : img
      )
    );
  };
  const handleShow = () => {
    setItems((prev) => prev.map((x) => ({ ...x, visible: true })));
  };

  const handleHide = () => {
    setItems((prev) => prev.map((x) => ({ ...x, visible: false })));
  };
  return (
    <div className="layout align-items-center mt-100">
      <div className="card ma-20 pa-50">
        <section>
          <div
            className="layout-row justify-content-around"
            data-testid="images-div"
          >
            {items.map((img, index) =>
              img.visible ? (
                <img
                  key={img.id}
                  src={img.src}
                  height={200}
                  width={300}
                  alt={img.all}
                  onClick={() => imgToggle(index)}
                />
              ) : (
                <HiddenImageDiv key={img.id} onClick={() => imgToggle(index)} />
              )
            )}
          </div>
        </section>
        <section className="card-actions justify-content-center">

		const stageId = task.stage;
		stagesTasks[stageId].push(task);
	}

	const [newTask, setNewTask] = useState('');
	const createTask = () => {
		if (newTask == "") {
			return;
		}
		const created = [...tasks, { name: newTask, stage: 0 }]
		setTasks(created);
		setNewTask("");

	}

		setTasks(tasks.map((t) => name == t.name ? ({ ...t, stage: t.stage + 1 }) : t));

	const backward = (name) => {
		setTasks(tasks.map((t) => name == t.name ? ({ ...t, stage: t.stage - 1 }) : t))
	}

	const handledelete = (name) => {
		setTasks(tasks.filter((d) => name != d.name))

function MovieList() {
  const [input, setInput] = useState();
  const [movies, setMovies] = useState([]);
  const [searched, setSearched] = useState(false);

  const search = async () => {
    const year = Number(input.trim());
    if (!year) return;
    try {
      const data = await fetch(
        `https://jsonmock.hackerrank.com/api/moviesdata?Year=${year}`
      );
      const json = await data.json();

      const arr = Array.isArray(json?.data) ? json.data : [];
      setMovies(arr);
    } catch {
      setMovies([]);
    } finally {
      setSearched(true);
    }
  };

  return (
    <div className="layout-column align-items-center mt-50">
      <section className="layout-row align-items-center justify-content-center">
        <input
          type="number"
          className="large"
          placeholder="Enter Year eg 2015"
          data-testid="app-input"
          value={input}
          onChange={(e) => setInput(e.target.value)}
        />
        <button onClick={search} className="" data-testid="submit-button">
          Search
        </button>
      </section>

      <ul className="mt-50 styled" data-testid="movieList">
        {movies.map((movie) => (
          <li key={movie.imdbId} className="slide-up-fade-in py-10">
            {movie.Title}
          </li>
        ))}

        <div className="mt-50 slide-up-fade-in" data-testid="no-result">

import React, { useEffect, useState } from "react";
import "./index.css"
export default function Timer({initial}) {
	const [initialValue, setInitialValue] = useState(initial);
	const [stopped, setStopped] = useState(false);
	useEffect(() => {
		if (stopped) {
			return;
		}
		const id = setInterval(() => {
			setInitialValue(prev => {
				if (prev <= 0) {
					clearInterval(id);
					setStopped(true);
					return 0;
				}
				return prev - 1;
			});
		}, 1000	);
		return () => clearInterval(id);
	}, [!stopped]);


	const handleStop = () => {
		setStopped(true);
	}
	return (
		<div className="mt-100 layout-column align-items-center justify-content-center">
			<div className="timer-value" data-testid="timer-value" >{initialValue}</div>
			<button className="large" data-testid="stop-button" onClick={handleStop}>Stop Timer</button>
		</div>
	);

}

    super(props);

      tasks: this.props.tasks,

    this.stagesNames = ["Backlog", "To Do", "Ongoing", "Done"];

    //CHANGEEEEE
  move = (name, dir) => {
    const newstage =
      [...this.state.tasks].find((e) => e.name === name).stage + dir;
    const newobj = { name: name, stage: newstage };
    const filteredtasks = [...this.state.tasks].filter((e) => e.name !== name);
    const finaltasks = [...filteredtasks, newobj];

class App extends Component {
    constructor() {
        super();
        const products = [...PRODUCTS].map((product, index) => {
            product.id = index + 1;
            product.image = `/images/items/${product.name.toLocaleLowerCase()}.png`;
            product.cartQuantity = 0;
            return product;
        });
        this.state = {
            cart: {
                items: [],
                subTotal: 0,
                totalPrice: 0,
                discount: 0,
                selectedCoupon: '0'
            },
            products
        }
        this.addToCart = this.addToCart.bind(this);
        this.removeFromCart = this.removeFromCart.bind(this);
    }

    addToCart(index) {
        const products = this.state.products;
        products[index].cartQuantity = 1;
        let cart = {...this.state.cart};
        cart.items.push({
            id: products[index].id,
            price: products[index].price,
            item: products[index].heading,
            quantity: 1
        });
        cart.subTotal= cart.subTotal + products[index].price;

        const discount= parseInt(cart.selectedCoupon);
        const calcDiscount= parseFloat((cart.subTotal * discount)/100);
        cart.discount= calcDiscount;

        const totalPr= cart.subTotal - cart.discount;
        cart.totalPrice= totalPr;
        this.setState({
            products,
            cart
        })
    }

    removeFromCart(index) {
        const products = this.state.products;
        products[index].cartQuantity = 0;
        let cart = {...this.state.cart};
        let cartIndex = this.state.cart.items.findIndex(item => item.id === products[index].id);
        cart.items.splice(cartIndex, 1);

        cart.subTotal= cart.subTotal - products[index].price

        const discount= parseInt(cart.selectedCoupon);
        const calcDist= parseFloat((cart.subTotal * discount)/100);
        cart.discount= calcDist;

        const totalPr= cart.subTotal - cart.discount;
        cart.totalPrice= totalPr;
        this.setState({
            cart,
            products
        })
    }

    addCoupon=(e)=>{
        const newCart= {...this.state.cart};
        const val= e.target.value;
        newCart.selectedCoupon= val;

        const discount= parseInt(val);
        const calcDist= parseFloat((newCart.subTotal * discount)/100);
        newCart.discount= calcDist;

        newCart.totalPrice= newCart.subTotal - newCart.discount

        this.setState({
            cart: newCart
        })
    }


    render() {

        return (
            <div>
                <h8k-navbar header={title}></h8k-navbar>
                <div className="layout-row shop-component">
                    <ProductList products={this.state.products} addToCart={this.addToCart} removeFromCart={this.removeFromCart}/>
                    <Cart cart={this.state.cart} addCoupon={this.addCoupon}/>
                </div>
            </div>
        );
    }
}

import React from 'react';
function Plaintext({ shift, plainText, setPlainText, setCipherText }) {
  
  const encryptText = (plainText, shift) => {
    let cipherArr = []
    let cipherLetter

    plainText.split("").map(letter => {
      let code = letter.charCodeAt(letter)
      if (letter === " ") {
        return cipherArr.push(letter)
      }
      // Uppercase letters
      if (code >= 65 && code <= 90) {
        cipherLetter = String.fromCharCode(((code - 65 + shift) % 26) + 65)
      }
      // Lowercase letters
      else if (code >= 97 && code <= 122) {
        cipherLetter = String.fromCharCode(((code - 97 + shift) % 26) + 97)
      }
      return cipherArr.push(cipherLetter)
    })
    return cipherArr.join("")
  }

  return (
    <div className="card w-18 mr-20">
      <h3 className="card-text text-center my-0 py-20">Plaintext</h3>
      <textarea
        name="plaintext"
        rows={6}
        cols={30}
        placeholder="Enter Plaintext"
        className="mx-25 mb-25"
        data-testid="plain-text"
        value={plainText}
        onChange={e => {
          setPlainText(e.target.value)
          const cipher = encryptText(e.target.value, Number(shift))
          setCipherText(cipher)
        }
        }
      />
    </div>
  )
}

import React from 'react'

function Ciphertext({shift, cipherText, setCipherText, setPlainText}) {
  
   const decryptText = (cipherText, shift) => {
    let plainArr = []
    let plainLetter

    cipherText.split("").map(cipher => {
      let code = cipher.charCodeAt(cipher)
      if (cipher === " ") {
        return plainArr.push(cipher)
      }
      // Uppercase letters
      if (code >= 65 && code <= 90) {
        let diff = code - 65 - shift
        if (diff >= 0) {
          plainLetter = String.fromCharCode((diff % 26) + 65);
        } else {
          plainLetter = String.fromCharCode(((26 + diff) % 26) + 65);
        }
      }
      // Lowercase letters
      else if (code >= 97 && code <= 122) {
        let diff = code - 97 - shift
        if (diff >= 0) {
          plainLetter = String.fromCharCode((diff % 26) + 97);
        } else {
          plainLetter = String.fromCharCode(((26 + diff) % 26) + 97);
        }
      }
      return plainArr.push(plainLetter)
    })
    return plainArr.join("")
  }

  return (
    <div className="card w-18 ml-20">
      <h3 className="card-text text-center my-0 py-20">Ciphertext</h3>
      <textarea
        name="ciphertext"
        rows={6}
        cols={30}
        placeholder="Enter Ciphertext"
        className="mx-25 mb-25"
        data-testid="cipher-text"
        value={cipherText}
        onChange={e => {
          setCipherText(e.target.value)
          const plain = decryptText(e.target.value, Number(shift))
          setPlainText(plain)
        }
        }
      />
    </div>
  )
}

  let months = ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"];
  const [cm, setCm] = useState("January"); // Current Month
  const [filteredStocks, setFilteredStocks] = useState([...stocks].filter((s) => s.date.includes("January")));
  function updateStocks(currentMonth){
    setCm(currentMonth);
    setFilteredStocks(stocks.filter((s) => s.date.includes(currentMonth)));
  }
  function handleSort() {
    setFilteredStocks([...filteredStocks].sort((a,b)=>(b.close-b.open)-(a.close-a.open)));
  }
  function calTotalProfit() {
    let profit = 0;
    for (let index = 0; index < filteredStocks.length; index++) {
      let p = filteredStocks[index].close-filteredStocks[index].open;
      if(p>0){
        profit = profit + p; 
      }
    }
    return Math.abs(profit).toFixed(2);
  }
  function calTotalLoss() {
    let loss = 0;
    for (let index = 0; index < filteredStocks.length; index++) {
      let p = filteredStocks[index].close-filteredStocks[index].open;
      if(p<0){
        loss = loss + p; 
      }
    }
    return Math.abs(loss).toFixed(2);
  }
  return (
    <>
      <div className='layout-row justify-content-center'>
        <select className='mt-13 mr-5' data-testid="select-month" value={cm} onChange={(e) => {updateStocks(e.target.value);}}>
          {
            months.map((month) => (
              <option value={month} key={month}>{month}</option>
            ))
          }
        </select>
        <button data-testid="sort" onClick={()=>handleSort()}>Sort</button>
      </div>
      <div className='layout-row wrap w-100 justify-content-center' data-testid="stocks">
        {
          filteredStocks.map((s) =>
          (
            <div className='card layout-column w-20 ma-10 px-8'>
              <p className='ma-0'>Date: {s.date}</p>
              <p className='ma-0'>High: {s.high}</p>
              <p className='ma-0'>Low: {s.low}</p>
              <p className='ma-0'>Profit: {(s.close - s.open)>0?"✅":"❌"} ({(Math.abs(s.close - s.open)).toFixed(2)})</p>
            </div>
          )
          )
        }

      </div>
      <div className='layout-column justify-content-center align-items-center' >
        <p className='mb-0 font-weight-bold' data-testid='total-profit'>Total Profit: <span style={{ color: "#1ba94c" }}>{calTotalProfit()}</span></p>
        <p className='font-weight-bold' data-testid='total-loss'>Total Loss: <span style={{ color: "red" }}>{calTotalLoss()}</span></p>
      </div>
    </>
  )
}

export default Users

const title = 'Editable Table'

const App = () => {
  const [employeesList, setEmployeeList] = useState([
    { id: 0, name: 'Chris Hatch', position: 'Software Developer', salary: 130000 },
    { id: 1, name: 'Elizabeth Montgomery', position: 'Lead Research Engineer', salary: 70000 },
    { id: 2, name: 'Aiden Shaw', position: 'Machine Learning Engineer', salary: 80000 },
  ])
  const [name, setName] = useState("");
  const [position, setPosition] = useState("");
  const [salary, setSalary] = useState("");

    setEmployeeList([...employeesList, { id: employeesList.length, name: name, position: position, salary: salary }]);
    setSalary("");
    setName("");
    setPosition("");
  }
  function updateEmployee(employee, newSalary) {
    setEmployeeList(prev => prev.map((emp) => emp === employee ? { ...emp, salary: newSalary } : emp))
  }

  return (
    <Fragment>
      <h8k-navbar header={title}></h8k-navbar>
      <div className="card w-45 mx-auto mt-75 pb-5">

          <thead>
            <tr>
              <th>Name</th>
              <th>Position</th>
              <th>Salary</th>
              <th>Action</th>
            </tr>
          </thead>
          <tbody>
            {employeesList.map((employee, idx) => (
              <tr key={employee.id} data-testid={`row-${idx}`}>
                <Employee
                  employee={employee}
                  idx={idx}
                  updateEmployee={updateEmployee}
                />
              </tr>
            ))}

  const [exp, setExp] = useState("");

import React, { useState } from "react";
import { CirclePlus, Trash2 } from "lucide-react";
import { toast } from "react-toastify";


function Tasks() {
  const [taskList, setTaskList] = useState([]);
  const [title, setTitle] = useState("");
  const [priority, setPriority] = useState(0);
  function addToTask() {
    if (title == "" || priority == "") {
      toast.error("Please enter both title and task priority");
      return;
    }
    let newTaskList = [...taskList, { title: title, priority: priority }];
    let sortedList = newTaskList.sort((a, b) => a.priority - b.priority);
    setTaskList(sortedList);
    setTitle("");
    setPriority(0);
  }
  function remove(t) {
    setTaskList(taskList.filter((task) => task !== t));
  }
  return (
    <div className="flex flex-col items-center w-full px-8">
      <div className="w-full max-w-5xl bg-white rounded-2xl shadow p-5 mt-12 mb-8">
        <div className="flex flex-row space-x-3 mb-6">
          <div className="bg-[#DCFAE6] text-[#18A149] rounded-lg p-4 flex items-center justify-center">
            <CirclePlus size={24} />
          </div>
          <div>
            <div className="font-bold text-md sm:text-lg">Create a new task</div>
            <div className="text-neutral-500 text-sm sm:text-base">
              Add task title and priority to create a new task
            </div>
          </div>
        </div>
        <form className="flex flex-col md:flex-row gap-6 mb-3">
          <input
            data-testid="input-task-name"
            className="flex-1 rounded-lg border border-gray-200 px-4 py-3 text-base focus:outline-none focus:ring-2 focus:ring-gray-800 bg-gray-100"
            placeholder="Enter task title"
            type="text"
            value={title}
            onChange={(e) => setTitle(e.target.value)}
          />
          <input
            data-testid="input-task-priority"

            value={priority}

          />
          <button
            data-testid="submit-button"
            className="h-12 px-8 rounded-md bg-black text-white font-bold text-base hover:bg-gray-900 transition"
            type="button"
            onClick={() => addToTask()}
          >
            Add
          </button>
        </form>
      </div>

      <div className="w-full max-w-5xl mb-6">
        <h2 className="text-3xl font-bold text-black">Your Tasks</h2>
      </div>
      <div className="w-full max-w-5xl bg-white rounded-2xl shadow overflow-hidden mb-8">
        <table className="w-full text-left">

import React from 'react';
function Plaintext({ shift, plainText, setPlainText, setCipherText }) {
  
  const encryptText = (plainText, shift) => {
    let cipherArr = []
    let cipherLetter

    plainText.split("").map(letter => {
      let code = letter.charCodeAt(letter)
      if (letter === " ") {
        return cipherArr.push(letter)
      }
      // Uppercase letters
      if (code >= 65 && code <= 90) {
        cipherLetter = String.fromCharCode(((code - 65 + shift) % 26) + 65)
      }
      // Lowercase letters
      else if (code >= 97 && code <= 122) {
        cipherLetter = String.fromCharCode(((code - 97 + shift) % 26) + 97)
      }
      return cipherArr.push(cipherLetter)
    })
    return cipherArr.join("")
  }

  return (
    <div className="card w-18 mr-20">
      <h3 className="card-text text-center my-0 py-20">Plaintext</h3>
      <textarea
        name="plaintext"
        rows={6}
        cols={30}
        placeholder="Enter Plaintext"
        className="mx-25 mb-25"
        data-testid="plain-text"
        value={plainText}
        onChange={e => {
          setPlainText(e.target.value)
          const cipher = encryptText(e.target.value, Number(shift))
          setCipherText(cipher)
        }
        }
      />
    </div>
  )
}

import React from 'react'

function Ciphertext({shift, cipherText, setCipherText, setPlainText}) {
  
   const decryptText = (cipherText, shift) => {
    let plainArr = []
    let plainLetter

    cipherText.split("").map(cipher => {
      let code = cipher.charCodeAt(cipher)
      if (cipher === " ") {
        return plainArr.push(cipher)
      }
      // Uppercase letters
      if (code >= 65 && code <= 90) {
        let diff = code - 65 - shift
        if (diff >= 0) {
          plainLetter = String.fromCharCode((diff % 26) + 65);
        } else {
          plainLetter = String.fromCharCode(((26 + diff) % 26) + 65);
        }
      }
      // Lowercase letters
      else if (code >= 97 && code <= 122) {
        let diff = code - 97 - shift
        if (diff >= 0) {
          plainLetter = String.fromCharCode((diff % 26) + 97);
        } else {
          plainLetter = String.fromCharCode(((26 + diff) % 26) + 97);
        }
      }
      return plainArr.push(plainLetter)
    })
    return plainArr.join("")
  }

  return (
    <div className="card w-18 ml-20">
      <h3 className="card-text text-center my-0 py-20">Ciphertext</h3>
      <textarea
        name="ciphertext"
        rows={6}
        cols={30}
        placeholder="Enter Ciphertext"
        className="mx-25 mb-25"
        data-testid="cipher-text"
        value={cipherText}
        onChange={e => {
          setCipherText(e.target.value)
          const plain = decryptText(e.target.value, Number(shift))
          setPlainText(plain)
        }
        }
      />
    </div>
  )
}

export default Ciphertext

import React, { useEffect, useRef, useState } from "react";
import "./index.css"

export default function Timer({ initial }) {
	const [value, setValue] = useState(initial);
	const intRef = useRef(null);
	useEffect(() => {
		intRef.current = setInterval(
			() => { setValue((prev) => prev > 1 ? prev - 1 : 0); }, 1000);
		return () => clearInterval(intRef.current);
	}, []);
	function stop() {
		clearInterval(intRef.current);
	}
	return (
		<div className="mt-100 layout-column align-items-center justify-content-center">
			<div className="timer-value" data-testid="timer-value">{value}</div>
			<button className="large" onClick={stop} data-testid="stop-button">Stop Timer</button>
		</div>
	);

}
